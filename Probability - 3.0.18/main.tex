\begin{frame}{Question}
Let $X_1, X_2$, $\cdots$ be a sequence of independent and identically distributed chi-square
random variables, each having 4 degrees of freedom. Define
\begin{align*}
    S_n = \sum_{i=1}^{n} X_i
\end{align*}
$
\text{If } \frac{S_n}{n} \to \mu \text{ as } n \to \infty,\ \text{then } \mu =
$
\quad
 \begin{enumerate}

\item   $8$ 
\item   $16$
 \item $24$
\item  $32$
 \end{enumerate}
 \hfill(MA 2007)
\end{frame}
\begin{frame}{Solution}
A random variable  \textbf{X} is said to follow a Chi-square distribution with n  degrees of freedom if :
\begin{align}
f(x) &= \frac{1}{2^{n/2}\,\Gamma(n/2)} \, x^{\frac{n}{2} - 1} e^{-x/2} , \quad x > 0 \\
f(x) &= 0, \quad \text{otherwise}
\end{align}

\begin{align}
\mathbf{X} \sim \chi^2(n)
\end{align}
(This is special case of Gamma Distribution)
\end{frame}
\begin{frame}{Solution}
\begin{align}
\text{Given: } X \sim \chi^2(n), \quad 
f(x) = \frac{1}{2^{n/2}\,\Gamma(n/2)} x^{\frac{n}{2}-1} e^{-x/2}, \; x>0
\end{align}

\begin{align}
E[X] &= \int_{0}^{\infty} x f(x)\,dx \\
&= \int_{0}^{\infty} x \cdot \frac{1}{2^{n/2}\,\Gamma(n/2)} x^{\frac{n}{2}-1} e^{-x/2}\,dx \\
&= \frac{1}{2^{n/2}\,\Gamma(n/2)} \int_{0}^{\infty} x^{\frac{n}{2}} e^{-x/2}\,dx
\end{align}

\begin{align}
\text{Let } t = \frac{x}{2} \;\Rightarrow\; x = 2t,\; dx = 2\,dt
\end{align}

\begin{align}
E[X] &= \frac{1}{2^{n/2}\,\Gamma(n/2)} 
\int_{0}^{\infty} (2t)^{\frac{n}{2}} e^{-t} \cdot 2\,dt 
\end{align}
\end{frame}


\begin{frame}{Solution}
\begin{align}
&= \frac{1}{2^{n/2}\,\Gamma(n/2)} 
\int_{0}^{\infty} 2^{n/2} t^{\frac{n}{2}} e^{-t} \cdot 2\,dt \\
&= \frac{2^{n/2} \cdot 2}{2^{n/2}\,\Gamma(n/2)} 
\int_{0}^{\infty} t^{\frac{n}{2}} e^{-t}\,dt \\
&= \frac{2}{\Gamma(n/2)} 
\int_{0}^{\infty} t^{\frac{n}{2}} e^{-t}\,dt
\end{align}
\begin{align}
\text{Using Gamma function: } 
\int_{0}^{\infty} t^{m-1} e^{-t} dt = \Gamma(m)
\end{align}
\begin{align}
\Rightarrow \int_{0}^{\infty} t^{\frac{n}{2}} e^{-t} dt 
= \Gamma\!\left(\frac{n}{2} + 1\right)
\end{align}
\end{frame}
\begin{frame}{Solution}


\begin{align}
E[X] &= \frac{2}{\Gamma(n/2)} \cdot \Gamma\!\left(\frac{n}{2} + 1\right)
\end{align}

\begin{align}
\text{Using } \Gamma(m+1) = m\,\Gamma(m): \\
\quad \Gamma\!\left(\frac{n}{2} + 1\right) 
= \frac{n}{2}\,\Gamma\!\left(\frac{n}{2}\right)
\end{align}

\begin{align}
E[X] &= \frac{2}{\Gamma(n/2)} \cdot \frac{n}{2}\,\Gamma\!\left(\frac{n}{2}\right) \\
&= n
\end{align}

\begin{align}
\boxed{E[X] = n}
\end{align}
\end{frame}

\begin{frame}{Solution}
According to our question , 
We have \\
\begin{align}
X_1, X_2, \ldots \sim \chi^2(4) \quad \text{(independent and identically distributed)}
\end{align}
Also from (19) , $E[X_i] = 4$
 Using \textbf{Law of Large Numbers(LLN)} , which states that the sample mean converges to the population mean as $n \to \infty$

 \begin{align}
\frac{1}{n}\sum_{i=1}^n X_i \xrightarrow{} {E}[X_i] \quad \text{as } n \to \infty
 \end{align}
\end{frame}



\begin{frame}{Solution}
Thus ,
\begin{align}
  \frac{S_n}{n} \xrightarrow{} {E}[X_i] \quad \text{as } n \to \infty\\
 \Rightarrow \frac{S_n}{n} \xrightarrow{} 4 \quad \text{as } n \to \infty
\end{align}
Therefore , $\mu = 4 $ (none of the options)
\end{frame}

import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import chi2

# Parameters
n_max = 5000
df = 4  

# Generate samples
X = np.random.chisquare(df, n_max)

#Plotting
plt.figure()
plt.hist(X, bins=50, density=True)
plt.title("X ~ Chi-square(4)")
plt.xlabel("x")
plt.ylabel("Density")
plt.grid(True)





\end{lstlisting}

S_n = np.cumsum(X)
n = np.arange(1, n_max + 1)
Sn_by_n = S_n / n

# Using scipy to get mean
true_mean = chi2.mean(df)

#Plotting
plt.figure()

plt.axhline(y=true_mean, linestyle='--', label="Mean")
plt.title("S_n / n vs n")
plt.xlabel("n")
plt.ylabel("S_n / n")
